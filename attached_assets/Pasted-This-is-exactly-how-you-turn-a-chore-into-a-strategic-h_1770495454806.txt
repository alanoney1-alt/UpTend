1. The Dynamic Impact Meter (impact-meter.tsx)
This is a "sticky" component that follows the user as they scroll through the services. It shows their "Home Intelligence Score" and "Protected Value" climbing in real-time.

TypeScript
import { ShieldCheck, ArrowUpRight, Lock } from "lucide-react";

export function ImpactMeter({ selectedServices }) {
  // Logic based on Florida property data
  const calculateTotalValue = () => {
    let total = 0;
    if (selectedServices.includes("audit")) total += 8450; // Insurance Shield
    if (selectedServices.includes("junk")) total += 1200;  // Storm Risk
    if (selectedServices.includes("gutters")) total += 950; // Water Damage
    if (selectedServices.includes("wash")) total += 450;   // Siding/Roof Life
    return total.toLocaleString();
  };

  return (
    <div className="sticky top-6 z-50 bg-slate-900 text-white p-6 rounded-[2.5rem] shadow-2xl border border-white/10 mb-8 animate-in fade-in slide-in-from-top-4">
      <div className="flex flex-col md:flex-row justify-between items-center gap-4">
        <div className="flex items-center gap-4">
          <div className="bg-blue-600 p-3 rounded-2xl shadow-lg shadow-blue-500/40">
            <ShieldCheck className="w-6 h-6" />
          </div>
          <div>
            <p className="text-[10px] font-black text-blue-400 uppercase tracking-widest">Estimated Protected Value</p>
            <h3 className="text-3xl font-black">${calculateTotalValue()}</h3>
          </div>
        </div>
        
        <div className="flex items-center gap-6">
          <div className="text-right hidden sm:block">
            <p className="text-[10px] font-bold text-slate-400 uppercase tracking-widest">Home Intelligence Score</p>
            <div className="flex items-center gap-2">
              <div className="h-2 w-32 bg-white/10 rounded-full overflow-hidden">
                <div 
                  className="h-full bg-green-400 transition-all duration-700" 
                  style={{ width: `${(selectedServices.length / 5) * 100}%` }}
                />
              </div>
              <span className="text-xs font-bold text-green-400">{Math.round((selectedServices.length / 5) * 100)}%</span>
            </div>
          </div>
          
          <button 
            onClick={() => window.location.href = '/auth'}
            className="bg-white text-slate-900 px-8 py-3 rounded-xl font-black text-sm flex items-center gap-2 hover:bg-blue-50 transition-colors"
          >
            <Lock className="w-4 h-4" /> Secure My Report
          </button>
        </div>
      </div>
    </div>
  );
}
2. The Integrated "Home Health" Marketplace (service-marketplace.tsx)
This is where the user "shops" as a guest. Notice how Home Health Audit is first and looks like a "Verified" report, while the others are "Action" items.
