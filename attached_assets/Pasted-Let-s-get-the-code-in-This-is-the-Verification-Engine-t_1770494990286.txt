import React, { useState } from "react";
import { Camera, Scale, Leaf, CheckCircle, ArrowRight, AlertCircle } from "lucide-react";
import { Button } from "@/components/ui/button";
import { Card, CardContent } from "@/components/ui/card";
import { Input } from "@/components/ui/input";

export function ProVerification({ jobId = "8829" }) {
  const [weight, setWeight] = useState("");
  const [material, setMaterial] = useState("Metal");
  const [photoTaken, setPhotoTaken] = useState(false);

  // Real-time impact calculation (Intelligence Pillar)
  const co2Avoided = (parseFloat(weight) || 0) * 1.5; 

  return (
    <div className="max-w-md mx-auto bg-slate-50 min-h-screen pb-20">
      {/* HEADER: CONTEXT */}
      <div className="bg-slate-900 text-white p-6 rounded-b-[2rem] shadow-xl">
        <div className="flex justify-between items-center mb-4">
          <span className="text-blue-400 text-[10px] font-black uppercase tracking-widest">Verification Mode</span>
          <span className="text-white/40 text-[10px]">Job #{jobId}</span>
        </div>
        <h2 className="text-2xl font-black">Final Impact Log</h2>
        <p className="text-slate-400 text-xs mt-1">Every entry here builds your Sustainability Badge.</p>
      </div>

      <div className="p-4 space-y-6 -mt-6">
        {/* 1. THE PHOTO PROOF (Accountability) */}
        <Card className="rounded-3xl border-none shadow-lg overflow-hidden">
          <CardContent className="p-0">
            <div 
              className={`aspect-video flex flex-col items-center justify-center transition-colors ${photoTaken ? 'bg-green-500' : 'bg-slate-200'}`}
              onClick={() => setPhotoTaken(true)}
            >
              {photoTaken ? (
                <div className="text-center text-white">
                  <CheckCircle className="w-12 h-12 mx-auto mb-2" />
                  <p className="font-bold">Receipt/Empty Space Logged</p>
                </div>
              ) : (
                <div className="text-center text-slate-400">
                  <Camera className="w-12 h-12 mx-auto mb-2" />
                  <p className="text-sm font-bold">Snap Receipt or Empty Space</p>
                  <p className="text-[10px]">Mandatory for Insurance Shield</p>
                </div>
              )}
            </div>
          </CardContent>
        </Card>

        {/* 2. MATERIAL DATA (Intelligence) */}
        <div className="space-y-4">
          <label className="text-xs font-black text-slate-400 uppercase tracking-widest ml-2">Material Details</label>
          <div className="grid grid-cols-2 gap-3">
            {["Metal", "Cardboard", "E-Waste", "Furniture"].map((m) => (
              <button
                key={m}
                onClick={() => setMaterial(m)}
                className={`p-4 rounded-2xl font-bold text-sm transition-all border-2 ${
                  material === m 
                  ? 'bg-blue-600 border-blue-600 text-white shadow-lg' 
                  : 'bg-white border-transparent text-slate-600'
                }`}
              >
                {m}
              </button>
            ))}
          </div>

          <div className="relative">
            <Scale className="absolute left-4 top-1/2 -translate-y-1/2 text-slate-400 w-5 h-5" />
            <Input 
              type="number" 
              placeholder="Enter Weight (lbs)" 
              className="h-14 pl-12 rounded-2xl border-none shadow-md text-lg font-bold"
              value={weight}
              onChange={(e) => setWeight(e.target.value)}
            />
          </div>
        </div>

        {/* 3. IMPACT PREVIEW (Gamification/Reward) */}
        <div className="bg-green-600 p-6 rounded-[2rem] text-white shadow-xl flex items-center justify-between">
          <div className="flex items-center gap-4">
            <div className="bg-white/20 p-3 rounded-xl">
              <Leaf className="w-6 h-6" />
            </div>
            <div>
              <p className="text-[10px] font-bold opacity-80 uppercase tracking-tighter">Your Verified Impact</p>
              <p className="text-2xl font-black">{co2Avoided.toFixed(1)}kg CO2 Saved</p>
            </div>
          </div>
        </div>

        {/* 4. SUBMIT: TRIGGER THE LEDGER */}
        <Button 
          disabled={!photoTaken || !weight}
          className="w-full h-16 rounded-2xl bg-slate-900 hover:bg-slate-800 text-white font-black text-lg shadow-2xl disabled:opacity-50"
        >
          Verify & Complete Job <ArrowRight className="ml-2 w-5 h-5" />
        </Button>
      </div>

      {/* COMPLIANCE WARNING */}
      <div className="px-8 flex gap-3 items-center opacity-60">
        <AlertCircle className="w-4 h-4 text-slate-400" />
        <p className="text-[9px] font-medium leading-tight">
          Fraudulent logging results in immediate suspension from the UpTend network. All data is cross-referenced with facility GPS and 360Â° video.
        </p>
      </div>
    </div>
  );
}
