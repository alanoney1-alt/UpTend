import { useRef } from "react";
import SignatureCanvas from "react-signature-canvas"; // Run: npm install react-signature-canvas
import { Button } from "@/components/ui/button";
import { Card } from "@/components/ui/card";

export function JobSignOff({ onSign }) {
  const sigPad = useRef({});

  const clear = () => sigPad.current.clear();
  
  const save = () => {
    if (sigPad.current.isEmpty()) return alert("Please sign first.");
    const dataUrl = sigPad.current.getTrimmedCanvas().toDataURL("image/png");
    onSign(dataUrl);
  };

  return (
    <Card className="p-4 bg-white space-y-4">
      <div className="text-center space-y-1">
        <h3 className="font-bold text-lg">Customer Sign-Off</h3>
        <p className="text-xs text-slate-500">
          I certify the job is complete and no damage occurred.
        </p>
      </div>

      <div className="border-2 border-dashed border-slate-300 rounded bg-slate-50">
        <SignatureCanvas 
          penColor="black"
          canvasProps={{width: 300, height: 150, className: 'sigCanvas mx-auto'}}
          ref={sigPad}
        />
      </div>

      <div className="flex gap-2">
        <Button variant="outline" onClick={clear} className="flex-1">Clear</Button>
        <Button onClick={save} className="flex-1 bg-green-600 hover:bg-green-700">
          Accept & Close Job
        </Button>
      </div>
    </Card>
  );
}

3. The "Ghost" Fee (Cancellation Logic)
The Risk: A competitor (or jerk) books 5 jobs to waste your drivers' time/fuel. The Fix: A coded Cancellation Policy.

üëâ Action: Update your Backend Logic (server/routes.ts).

// Logic to run when User clicks "Cancel Job"
app.post("/api/jobs/cancel", async (req, res) => {
  const { jobId } = req.body;
  const job = await db.query.orders.findFirst({ where: eq(orders.id, jobId) });
  
  const hoursUntilJob = (new Date(job.scheduledAt) - new Date()) / 36e5;
  
  let fee = 0;
  
  // 1. If Pycker is already "En Route" -> Charge $50
  if (job.status === 'en_route') {
    fee = 5000; // $50.00
  } 
  // 2. If < 24 hours notice -> Charge $25
  else if (hoursUntilJob < 24) {
    fee = 2500; // $25.00
  }

  if (fee > 0) {
    // Call Stripe to capture the "Hold" for the fee amount
    await stripe.paymentIntents.capture(job.paymentIntentId, { amount_to_capture: fee });
  } else {
    // Cancel the hold (Full Refund)
    await stripe.paymentIntents.cancel(job.paymentIntentId);
  }

  // Update DB status to 'cancelled'
});
