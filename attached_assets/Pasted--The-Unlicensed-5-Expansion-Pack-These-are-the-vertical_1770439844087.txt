2. The "Smart Pricing" Engine (server/services/pricing.ts)
Junk is priced by volume. Pressure washing is priced by square footage. Gutters are priced by linear feet (or story count).

We need a "Polymorphic Pricing" function.

TypeScript
// server/services/pricing.ts

export function calculateBasePrice(job: any) {
  switch (job.serviceType) {
    case 'junk_removal':
      return calculateVolumePrice(job.items); // Your existing AI logic

    case 'pressure_washing':
      // Market Rate: ~$0.20 - $0.30 per sq ft
      const sqFtRate = 25; // $0.25
      return job.squareFootage * sqFtRate;

    case 'gutter_cleaning':
      // Flat Rate: $150 (1 story) vs $250 (2 story)
      return job.storyCount === 1 ? 15000 : 25000;

    case 'moving_labor':
      // Hourly Rate: $50/hr per mover
      return job.laborHours * 5000 * 2; // (Assuming 2 movers)

    case 'light_demolition':
      // Custom AI Quote required (Use Vision Vault)
      return null; // Triggers "Get Custom Quote" flow
  }
}
