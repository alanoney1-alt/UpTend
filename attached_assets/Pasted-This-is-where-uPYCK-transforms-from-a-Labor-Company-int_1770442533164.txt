It separates "Our Jobs" from "Referral Jobs."File: client/src/components/consultation/consultation-wizard.tsxTypeScriptimport { useState } from "react";
import { Button } from "@/components/ui/button";
import { Card } from "@/components/ui/card";
import { Camera, CheckCircle, AlertTriangle, XCircle } from "lucide-react";

const CHECKLIST = [
  { id: 'roof', label: 'Roof Condition', type: 'referral', partner: 'Roofer' },
  { id: 'gutters', label: 'Gutters', type: 'upyck', price: 150 },
  { id: 'driveway', label: 'Driveway (Algae)', type: 'upyck', price: 99 },
  { id: 'hvac', label: 'HVAC Unit (Rust/Age)', type: 'referral', partner: 'HVAC Pro' },
  { id: 'garage', label: 'Garage Clutter', type: 'upyck', price: 250 },
  { id: 'trees', label: 'Overhanging Trees', type: 'referral', partner: 'Arborist' },
];

export function ConsultationWizard({ onComplete }) {
  const [results, setResults] = useState({});
  const [step, setStep] = useState(0);

  const currentItem = CHECKLIST[step];

  const handleRating = (status) => {
    setResults({ ...results, [currentItem.id]: status });
    if (step < CHECKLIST.length - 1) {
      setStep(step + 1);
    } else {
      finishConsultation();
    }
  };

  const finishConsultation = () => {
    // Separate into "uPYCK Jobs" and "Referrals"
    const upyckJobs = CHECKLIST.filter(i => results[i.id] === 'attention' && i.type === 'upyck');
    const referrals = CHECKLIST.filter(i => results[i.id] === 'critical' && i.type === 'referral');
    
    onComplete({ upyckJobs, referrals });
  };

  return (
    <div className="p-4 h-full bg-slate-50 flex flex-col justify-between">
      <div>
        <div className="flex justify-between items-center mb-6">
          <h2 className="text-xl font-bold">Health Check</h2>
          <span className="bg-slate-200 px-3 py-1 rounded-full text-sm">
            {step + 1} / {CHECKLIST.length}
          </span>
        </div>

        <Card className="p-8 text-center mb-8">
          <h1 className="text-3xl font-bold mb-2">{currentItem.label}</h1>
          <p className="text-slate-500">
            {currentItem.type === 'upyck' ? "Look for dirt/clutter." : "Look for visible damage."}
          </p>
        </Card>

        <div className="grid gap-4">
          <Button 
            className="h-16 bg-green-100 text-green-800 hover:bg-green-200 border border-green-300"
            onClick={() => handleRating('good')}
          >
            <CheckCircle className="mr-2" /> Looks Good
          </Button>

          {currentItem.type === 'upyck' ? (
            <Button 
              className="h-16 bg-yellow-100 text-yellow-800 hover:bg-yellow-200 border border-yellow-300"
              onClick={() => handleRating('attention')}
            >
              <AlertTriangle className="mr-2" /> Needs Cleaning (${currentItem.price})
            </Button>
          ) : (
             <Button 
              className="h-16 bg-red-100 text-red-800 hover:bg-red-200 border border-red-300"
              onClick={() => handleRating('critical')}
            >
              <XCircle className="mr-2" /> Needs Repair (Referral)
            </Button>
          )}
        </div>
      </div>
      
      <p className="text-center text-xs text-slate-400 mt-4">
        Take photos for any "Needs Cleaning" or "Repair" items.
      </p>
    </div>
  );
}
