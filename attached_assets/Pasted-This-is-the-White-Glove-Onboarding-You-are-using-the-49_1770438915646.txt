});
2. The AI Vision Engine (server/services/inventory-ai.ts)
This service takes the video (or a series of photos) from the consultation and turns it into a structured database.

TypeScript
// server/services/inventory-ai.ts
import OpenAI from "openai";

const openai = new OpenAI();

export async function parseInventoryFromVideo(videoUrl: string, userId: string) {
  // 1. The Prompt: "Watch this video and list every major asset."
  const prompt = `
    Analyze this home walkthrough video. Identify every distinct household item 
    (furniture, electronics, appliances, gym equipment) visible.
    
    For each item, provide:
    1. Name (Specific, e.g., "Peloton Bike+" or "Leather Sectional")
    2. Category
    3. Estimated Resale Value (Conservative US Dollars)
    4. Condition (Visual assessment)
    5. A brief description of visual state.

    Return as a JSON array. Ignore small clutter (books, clothes).
  `;

  // 2. Call GPT-4o Vision
  const response = await openai.chat.completions.create({
    model: "gpt-4o",
    messages: [
      { role: "system", content: "You are a professional home insurance adjuster." },
      { 
        role: "user", 
        content: [
          { type: "text", text: prompt },
          { type: "image_url", image_url: { url: videoUrl } } // OpenAI supports video frames or URLs
        ]
      }
    ],
    response_format: { type: "json_object" }
  });

  const inventoryList = JSON.parse(response.choices[0].message.content).items;

  // 3. Save to Database
  const savedItems = [];
  for (const item of inventoryList) {
    const [saved] = await db.insert(homeInventory).values({
      userId,
      itemName: item.name,
      category: item.category,
      estimatedValue: item.estimated_value * 100, // Store as cents
      condition: item.condition,
      conditionNotes: item.description,
      confidenceScore: 90, // Baseline for AI
    }).returning();
    savedItems.push(saved);
  }

  return savedItems;
}
