import React, { useState } from "react";
import { ShieldCheck, Zap, Leaf, MapPin, ArrowRight } from "lucide-react";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";

interface HeroProps {
  onAddressSelect: (address: string) => void;
}

export function Hero({ onAddressSelect }: HeroProps) {
  const [address, setAddress] = useState("");

  const handleScan = () => {
    if (address.length > 5) {
      onAddressSelect(address);
    }
  };

  return (
    <section className="relative pt-20 pb-32 overflow-hidden bg-white">
      {/* BACKGROUND ACCENTS */}
      <div className="absolute top-0 left-1/2 -translate-x-1/2 w-full h-full pointer-events-none z-0">
        <div className="absolute top-[-10%] left-[-10%] w-[40%] h-[40%] bg-blue-50 rounded-full blur-[120px] opacity-60" />
        <div className="absolute bottom-[-10%] right-[-10%] w-[40%] h-[40%] bg-green-50 rounded-full blur-[120px] opacity-60" />
      </div>

      <div className="relative z-10 max-w-7xl mx-auto px-6 text-center">
        {/* BRAND PILLARS TAGS */}
        <div className="flex justify-center gap-3 mb-8 animate-in fade-in slide-in-from-bottom-4 duration-700">
          <span className="flex items-center gap-2 px-4 py-2 bg-blue-50 text-blue-600 rounded-full text-xs font-black uppercase tracking-widest border border-blue-100">
            <ShieldCheck className="w-3 h-3" /> Protect
          </span>
          <span className="flex items-center gap-2 px-4 py-2 bg-cyan-50 text-cyan-600 rounded-full text-xs font-black uppercase tracking-widest border border-cyan-100">
            <Zap className="w-3 h-3" /> Connect
          </span>
          <span className="flex items-center gap-2 px-4 py-2 bg-green-50 text-green-600 rounded-full text-xs font-black uppercase tracking-widest border border-green-100">
            <Leaf className="w-3 h-3" /> Sustain
          </span>
        </div>

        {/* PRIMARY TAGLINE */}
        <h1 className="text-5xl md:text-7xl font-black text-slate-900 mb-6 tracking-tight leading-[1.1]">
          Home intelligence.<br />
          Instant service.<br />
          <span className="text-transparent bg-clip-text bg-gradient-to-r from-green-600 to-blue-500">
            Proven sustainability.
          </span>
        </h1>

        {/* SUBHEADLINE */}
        <p className="text-slate-500 text-lg md:text-xl max-w-3xl mx-auto mb-12 leading-relaxed font-medium">
          The first home services platform where every job is tracked, every pro is valued, and every impact is measured.
        </p>

        {/* ADDRESS INPUT (FRICTIONLESS FLOW) */}
        <div className="max-w-2xl mx-auto bg-white p-2 rounded-3xl shadow-2xl border border-slate-100 flex flex-col sm:flex-row items-center gap-2">
          <div className="flex-1 flex items-center px-4 gap-3 w-full">
            <MapPin className="text-blue-500 w-6 h-6 shrink-0" />
            <Input 
              placeholder="Enter your property address..." 
              className="border-none shadow-none focus-visible:ring-0 text-lg h-14 w-full"
              value={address}
              onChange={(e) => setAddress(e.target.value)}
              onKeyDown={(e) => e.key === 'Enter' && handleScan()}
            />
          </div>
          <Button 
            onClick={handleScan}
            className="w-full sm:w-auto bg-blue-600 hover:bg-blue-500 h-14 px-10 rounded-2xl text-lg font-bold shadow-xl shadow-blue-500/20 transition-all active:scale-95"
          >
            Scan Property <ArrowRight className="ml-2 w-5 h-5" />
          </Button>
        </div>

        {/* VALUE PROPOSITION FOOTNOTE */}
        <p className="mt-6 text-[10px] text-slate-400 font-bold uppercase tracking-[0.2em]">
          Free Home Intelligence Report ‚Ä¢ No Login Required

      </div>
    </section>
  );
}
